>               &500 <   x \le 1000, \\
>           0,  &x \not \in [100,1000] \\
>         \end{cases} \tag{1.1}
> \end{aligned}

> Determine the following:

cdf, from (1.1):

$$
\begin{align}
F(X = x) &= \int_{-\infty}^x f(u)du \notag \\
F(X = x) &= \begin{cases}
          0,  &x < 100, \\
          \int_{100}^x \big[-5.56 \times 10^{-4} + 5.56 \times 10^{-6}u\big]du,

### part (a)

:::{.question}

> P(X < 500).

:::

$$
\begin{align}
P(X < 500) &= F(500) \notag \\
           &= -2.22 \times 10^{-6} (500)^2 + 4.44 \times 10^{-3} (500) - 1.2202 &&
               \htmlClass{hljs-comment}{\textit{// case $100 \le x \le 500$ of (1.2)}} \notag \\
           &= 0.4448 \space_\blacksquare \tag{1.a}
\end{align}
$$

### part (b)

:::{.question}

> P(100 < X ≤ 200).

:::

$$
\begin{aligned}
P(100 < X ≤ 200)
    &= F(200) - F(100) \\
    &= F(200) &&
        \htmlClass{hljs-comment}{\textit{// $F(100) = 0$}} \\
    &= -2.22 \times 10^{-6} (100)^2 + 4.44 \times 10^{-3} (100) - 1.2202 &&
        \htmlClass{hljs-comment}{\textit{// case $100 \le x \le 500$ of (1.2)}} \\
    &= 0.0277\bar{9} \space_\blacksquare
\end{aligned}
$$

</section>
<section>

### part (c)

:::{.question}

> P(X > 800)

:::

$$
\begin{aligned}
P(X > 800)
    &= 1 - P(X \le 800) \\
    &= 1 - F(800) \\
    &= 1 - (-2.22 \times 10^{-6} x^2 + 4.44 \times 10^{-3} x - 1.2202) &&
        \htmlClass{hljs-comment}{\textit{// case $500 \lt x \le 1000$ of (1.2)}} \\
    &= 1 - 0.911 \\
    &= 0.089 \space_\blacksquare
\end{aligned}
$$

</section>

### part (d)

:::{.question}

> Find the power loss value such that only 10% of power losses are
> larger than this value.

:::

$$
\begin{aligned}
P(X > x) &= 0.10 = 1 - P(X \le x) \\
    0.10 &= 1 - F(x) \\
    0.90 &= F(x), 500 < x \lt 1000 &&
        \htmlClass{hljs-comment}{\textit{// 0.90 > F(500), by (1.a)}} \\
    0.90 &= -2.22 \times 10^{-6} x^2 + 4.44 \times 10^{-3} x - 1.2202, 500 < x \lt 1000 &&
        \htmlClass{hljs-comment}{\textit{// by case $500 \lt x \le 1000$ of (1.2)}} \\
       0 &= -2.22 \times 10^{-6} x^2 + 4.44 \times 10^{-3} x - 2.1202, 500 < x \lt 1000 \\
       x &\approx 787.97 \space_\blacksquare &&
        \htmlClass{hljs-comment}{\textit{// by quadratic formula}} \\
\end{aligned}
$$
